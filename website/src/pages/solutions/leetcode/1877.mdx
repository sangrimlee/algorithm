## Solution

```typescript filename="solution.ts" copy
export function minPairSum(nums: number[]): number {
  const n = nums.length;

  let answer = 0;
  nums.sort((a, b) => a - b);
  for (let i = 0; i < n / 2; i++) {
    answer = Math.max(answer, nums[i] + nums[n - i - 1]);
  }
  return answer;
}
```
