## Solution

```typescript filename="solution.ts" copy
export function areNumbersAscending(s: string): boolean {
  const nums = s.split(' ').filter((v) => !isNaN(parseInt(v)));
  return nums.every((num, i) => i === 0 || parseInt(nums[i - 1]) < parseInt(num));
}
```
