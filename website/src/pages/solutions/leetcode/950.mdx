## Solution

```typescript filename="solution.ts" copy
export function deckRevealedIncreasing(deck: number[]): number[] {
  deck.sort((a, b) => b - a);
  const queue: number[] = [];
  deck.forEach((card) => {
    if (0 < queue.length) {
      queue.push(queue.shift()!);
    }
    queue.push(card);
  });
  return queue.reverse();
}
```
