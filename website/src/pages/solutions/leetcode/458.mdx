## Solution

```typescript filename="solution.ts" copy
export function poorPigs(buckets: number, minutesToDie: number, minutesToTest: number): number {
  return Math.ceil(Math.log10(buckets) / Math.log10(minutesToTest / minutesToDie + 1));
}
```
