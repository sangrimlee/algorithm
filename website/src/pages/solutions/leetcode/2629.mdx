# 2629. Function Composition

<Badges>

</Badges>

## Solution

```typescript filename="solution.ts" copy
type F = (x: number) => number;

export function compose(functions: F[]): F {
  return (x) => functions.reduceRight((acc, func) => func(acc), x);
}
```
