## Solution

```typescript filename="solution.ts" copy
export function createTargetArray(nums: number[], indices: number[]): number[] {
  const answer: number[] = [];
  indices.forEach((index, numIndex) => {
    answer.splice(index, 0, nums[numIndex]);
  });
  return answer;
}
```
