## Solution

```typescript filename="solution.ts" copy
export function lengthOfLastWord(s: string): number {
  let [start, end] = [s.length - 1, -1];
  while (0 <= start && (s[start] !== ' ' || end < 0)) {
    if (s[start] !== ' ' && end < 0) {
      end = start;
    }
    start -= 1;
  }
  return end - start;
}
```
