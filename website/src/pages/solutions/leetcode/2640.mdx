## Solution

```typescript filename="solution.ts" copy
export function findPrefixScore(nums: number[]): number[] {
  const answer: number[] = [];
  let sum = 0;
  let maxValue = 0;
  for (const num of nums) {
    maxValue = Math.max(maxValue, num);
    sum += maxValue + num;
    answer.push(sum);
  }
  return answer;
}
```

### Complexity

- Time: $O(n)$
- Space: $O(n)$
