## Solution

```typescript filename="solution.ts" copy
export function countEven(num: number): number {
  let current = num;
  let digitSum = 0;
  while (0 < current) {
    digitSum += current % 10;
    current = Math.floor(current / 10);
  }
  return Math.floor((num - (digitSum % 2)) / 2);
}
```
