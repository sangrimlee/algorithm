## Solution

```typescript filename="solution.ts" copy
export function removePalindromeSub(s: string): number {
  function isPalindrome(s: string) {
    let [start, end] = [0, s.length - 1];
    while (start < end) {
      if (s[start] !== s[end]) {
        return false;
      }
      start++;
      end--;
    }
    return true;
  }
  return s.length ? (isPalindrome(s) ? 1 : 2) : 0;
}
```
