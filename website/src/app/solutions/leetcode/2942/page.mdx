## Solution

```typescript filename="solution.ts" copy
export function findWordsContaining(words: string[], x: string): number[] {
  const answer: number[] = [];
  words.forEach((word, i) => {
    if (word.includes(x)) {
      answer.push(i);
    }
  });
  return answer;
}
```
