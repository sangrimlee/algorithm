## Solution

```typescript filename="solution.ts" copy
export function constructTransformedArray(nums: number[]): number[] {
  const n = nums.length;
  return nums.map((num, i) => nums[(n + i + (num % n)) % n]);
}
```

### Complexity

- Time: $O(n)$
- Space: $O(n)$
