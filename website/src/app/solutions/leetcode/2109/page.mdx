## Solution

```typescript filename="solution.ts" copy
export function addSpaces(s: string, spaces: number[]): string {
  const words = [];
  let prevSpace = 0;
  for (const space of spaces) {
    words.push(s.substring(prevSpace, space));
    prevSpace = space;
  }
  words.push(s.substring(prevSpace));
  return words.join(' ');
}
```

### Complexity

- Time: $O(n)$
- Space: $O(n)$
