## Solution

```typescript filename="solution.ts" copy
export function differenceOfSums(n: number, m: number): number {
  let answer = 0;
  for (let num = 1; num <= n; num++) {
    answer += num % m !== 0 ? num : -num;
  }
  return answer;
}
```
