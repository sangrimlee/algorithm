## Solution

```typescript filename="solution.ts" copy
export function getLastMoment(n: number, left: number[], right: number[]): number {
  let result = 0;
  for (const i of left) {
    result = Math.max(result, i);
  }
  for (const i of right) {
    result = Math.max(result, n - i);
  }
  return result;
}
```
