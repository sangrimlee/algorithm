## Solution

```typescript filename="solution.ts" copy
import { TreeNode } from '@algorithm/lib';

export function inorderTraversal(root: TreeNode | null): number[] {
  const answer: number[] = [];

  function inorder(node: TreeNode | null) {
    if (node === null) {
      return;
    }
    inorder(node.left);
    answer.push(node.val);
    inorder(node.right);
  }

  inorder(root);
  return answer;
}
```
