## Solution

```typescript filename="solution.ts" copy
import { ListNode } from '@algorithm/lib';

export function getDecimalValue(head: ListNode | null): number {
  let answer = 0;
  let currentNode = head;
  while (currentNode) {
    answer <<= 1;
    answer += currentNode.val;
    currentNode = currentNode.next;
  }
  return answer;
}
```
