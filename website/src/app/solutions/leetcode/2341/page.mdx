## Solution

```typescript filename="solution.ts" copy
export function numberOfPairs(nums: number[]): number[] {
  const set = new Set<number>();

  let numOfPairs = 0;
  for (const num of nums) {
    if (set.has(num)) {
      set.delete(num);
      numOfPairs += 1;
    } else {
      set.add(num);
    }
  }

  return [numOfPairs, set.size];
}
```
