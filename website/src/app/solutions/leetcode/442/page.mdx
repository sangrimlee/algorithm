## Solution

```typescript filename="solution.ts" copy
export function findDuplicates(nums: number[]): number[] {
  const answer: number[] = [];
  nums.forEach((num) => {
    if (nums[Math.abs(num) - 1] < 0) {
      answer.push(Math.abs(num));
    } else {
      nums[Math.abs(num) - 1] *= -1;
    }
  });
  return answer;
}
```
