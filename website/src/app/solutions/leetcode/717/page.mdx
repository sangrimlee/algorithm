## Solution

```typescript filename="solution.ts" copy
export function isOneBitCharacter(bits: number[]): boolean {
  const n = bits.length;
  let currentIndex = 0;
  while (currentIndex < n) {
    if (currentIndex === n - 1) {
      return true;
    }
    currentIndex += bits[currentIndex] === 1 ? 2 : 1;
  }
  return false;
}
```
