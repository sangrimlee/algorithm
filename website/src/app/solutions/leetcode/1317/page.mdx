## Solution

```typescript filename="solution.ts" copy
export function getNoZeroIntegers(n: number): number[] {
  const hasZero = (num: number) => {
    while (0 < num) {
      if (num % 10 === 0) {
        return true;
      }
      num = Math.floor(num / 10);
    }
    return false;
  };

  let num = 1;
  while (hasZero(num) || hasZero(n - num)) {
    num += 1;
  }
  return [num, n - num];
}
```
