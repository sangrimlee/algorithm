## Solution

```typescript filename="solution.ts" copy
export function minMovesToSeat(seats: number[], students: number[]): number {
  seats.sort((a, b) => a - b);
  students.sort((a, b) => a - b);
  return seats.reduce((answer, seat, i) => answer + Math.abs(seat - students[i]), 0);
}
```
