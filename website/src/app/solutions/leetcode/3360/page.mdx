## Solution

```typescript filename="solution.ts" copy
export function canAliceWin(n: number): boolean {
  let remain = n;
  for (let turn = 0; turn <= 10; turn++) {
    if (remain < 10 - turn) {
      return turn % 2 !== 0;
    }
    remain -= 10 - turn;
  }
  return true;
}
```

### Compleixty

- Time: $O(1)$
- Space: $O(1)$
