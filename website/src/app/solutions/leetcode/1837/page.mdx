## Solution

```typescript filename="solution.ts" copy
export function sumBase(n: number, k: number): number {
  let answer = 0;
  let current = n;
  while (0 < current) {
    answer += current % k;
    current = Math.floor(current / k);
  }
  return answer;
}
```
