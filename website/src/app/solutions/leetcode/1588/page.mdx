## Solution

```typescript filename="solution.ts" copy
export function sumOddLengthSubarrays(arr: number[]): number {
  return arr.reduce((acc, num, i) => acc + num * Math.ceil(((i + 1) * (arr.length - i)) / 2), 0);
}
```
