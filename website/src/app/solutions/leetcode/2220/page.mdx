# 2220. Minimum Bit Flips to Convert Number

## Solution

```typescript filename="solution.ts" copy
export function minBitFlips(start: number, goal: number): number {
  const xor = start ^ goal;
  return xor.toString(2).replace(/0/g, '').length;
}
```
