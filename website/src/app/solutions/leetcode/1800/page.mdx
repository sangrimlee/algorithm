## Solution

```typescript filename="solution.ts" copy
export function maxAscendingSum(nums: number[]): number {
  let answer = 0;
  let lastNum = 0;
  let currentSum = 0;
  nums.forEach((num) => {
    currentSum = lastNum < num ? currentSum + num : num;
    lastNum = num;
    answer = Math.max(answer, currentSum);
  });
  return answer;
}
```
