# 2666. Allow One Function Call

<Badge.Root>
<Badge.Level level="easy" />

</Badge.Root>

## Solution

```typescript filename="solution.ts" copy
export function once<T, R>(fn: (...args: T[]) => R): (...args: T[]) => R | undefined {
  let called = false;
  return (...args) => {
    if (called) {
      return;
    }
    called = true;
    return fn(...args);
  };
}
```
