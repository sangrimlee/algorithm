## Solution

```typescript filename="solution.ts" copy
export function separateDigits(nums: number[]): number[] {
  function separateDigit(num: number): number[] {
    const digits: number[] = [];
    while (0 < num) {
      digits.push(num % 10);
      num = Math.floor(num / 10);
    }
    return digits;
  }
  return nums.flatMap(separateDigit);
}
```
