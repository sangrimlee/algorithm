## Solution

```typescript showLineNumbers
export function solution(elements: number[]) {
  const set = new Set();
  const extended = [...elements, ...elements];

  for (let i = 0; i < elements.length; i++) {
    let sumValue = 0;
    for (let j = 0; j < elements.length; j++) {
      sumValue += extended[i + j];
      set.add(sumValue);
    }
  }

  return set.size;
}
```
