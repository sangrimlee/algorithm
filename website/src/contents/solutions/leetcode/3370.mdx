## Solution

```typescript filename="solution.ts" copy
export function smallestNumber(n: number): number {
  return (1 << bitLength(n)) - 1;
}

function bitLength(num: number): number {
  let length = 0;
  while (0 < num) {
    num >>= 1;
    length += 1;
  }
  return length;
}
```

### Complexity

- Time: $O(logn)$
- Space: $O(1)$
