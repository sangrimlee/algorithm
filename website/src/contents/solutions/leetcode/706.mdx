## Solution

```typescript filename="solution.ts" copy
export class MyHashMap extends Map<number, number> {
  put(key: number, value: number): void {
    super.set(key, value);
  }

  get(key: number): number {
    return super.get(key) ?? -1;
  }

  remove(key: number): void {
    super.delete(key);
  }
}
```
