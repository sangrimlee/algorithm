## Solution

```typescript filename="solution.ts" copy
export function minimumOperations(nums: number[]): number {
  return nums.reduce((prev, num) => (num % 3 === 0 ? prev : prev + 1), 0);
}
```

### Complexity

- Time: `O(n)`
- Space: `O(1)`
