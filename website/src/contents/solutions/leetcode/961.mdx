## Solution

```typescript filename="solution.ts" copy
export function repeatedNTimes(nums: number[]): number {
  const set = new Set<number>();
  for (const num of nums) {
    if (set.has(num)) {
      return num;
    }
    set.add(num);
  }
  throw new Error('There is no repeated element.');
}
```
