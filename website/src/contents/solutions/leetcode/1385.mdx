## Solution

```typescript filename="solution.ts" copy
export function findTheDistanceValue(arr1: number[], arr2: number[], d: number): number {
  return arr1.filter((num1) => arr2.every((num2) => Math.abs(num1 - num2) > d)).length;
}
```
