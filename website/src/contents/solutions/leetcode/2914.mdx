## Solution

```typescript filename="solution.ts" copy
export function minChanges(s: string): number {
  const n = s.length;
  let answer = 0;
  for (let i = 0; i < n; i += 2) {
    if (s[i] !== s[i + 1]) {
      answer += 1;
    }
  }
  return answer;
}
```

### Complexity

- Time: $O(n)$
- Space: $O(1)$
