## Solution

```typescript filename="solution.ts" copy
export function countOdds(low: number, high: number): number {
  if (low % 2 === 1 || high % 2 === 1) {
    return Math.ceil((high - low + 1) / 2);
  }
  return (high - low) / 2;
}
```
