## Solution

```typescript showLineNumbers
export function reductionOperations(nums: number[]): number {
  const n = nums.length;
  nums.sort((a, b) => a - b);

  let answer = 0;
  let prevLargestCount = 0;
  for (let i = 1; i < n; i++) {
    if (nums[i - 1] !== nums[i]) {
      prevLargestCount += 1;
    }
    answer += prevLargestCount;
  }
  return answer;
}
```
