## Solution

```typescript filename="solution.ts" copy
export function maximumElementAfterDecrementingAndRearranging(arr: number[]): number {
  return arr.sort((a, b) => a - b).reduce((prev, num) => Math.min(prev + 1, num), 0);
}
```
