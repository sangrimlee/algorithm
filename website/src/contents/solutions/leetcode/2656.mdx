## Solution

```typescript showLineNumbers
export function maximizeSum(nums: number[], k: number): number {
  const maxNum = nums.reduce((max, num) => (max < num ? num : max), 0);
  return maxNum * k + (k * (k - 1)) / 2;
}
```
