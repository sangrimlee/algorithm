## Solution

```typescript showLineNumbers
export function minMoves2(nums: number[]): number {
  nums.sort((a, b) => a - b);
  const n = nums.length;
  const mid = nums[Math.floor(n / 2)];
  return nums.reduce((prev, curr) => prev + Math.abs(mid - curr), 0);
}
```
