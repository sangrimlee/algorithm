## Solution

```typescript filename="solution.ts" copy
export function checkString(s: string): boolean {
  const lastAIndex = s.lastIndexOf('a');
  const firstBIndex = s.indexOf('b');
  if (lastAIndex === -1 || firstBIndex === -1) {
    return true;
  }
  return lastAIndex < firstBIndex;
}
```
