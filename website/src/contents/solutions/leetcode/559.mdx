## Solution

```typescript showLineNumbers
import { NArrayTreeNode } from '@algorithm/lib';

export function maxDepth(root: NArrayTreeNode | null): number {
  if (!root) {
    return 0;
  }
  return 1 + root.children.reduce((prev, node) => Math.max(prev, maxDepth(node)), 0);
}
```
