## Solution

```typescript filename="solution.ts" copy
export function occurrencesOfElement(nums: number[], queries: number[], x: number): number[] {
  const indices: number[] = [];
  nums.forEach((num, i) => {
    if (num === x) {
      indices.push(i);
    }
  });

  return queries.map((query) => (query <= indices.length ? indices[query - 1] : -1));
}
```

### Complexity

- Time: `O(N+M)`
- Space: `O(N)`
