## Solution

```typescript filename="solution.ts" copy
export function countSeniors(details: string[]): number {
  function isSenior(detail: string) {
    return 60 < parseInt(detail.substring(11, 13), 10);
  }
  return details.filter(isSenior).length;
}
```
