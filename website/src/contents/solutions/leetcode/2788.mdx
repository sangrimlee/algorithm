## Solution

```typescript showLineNumbers
export function splitWordsBySeparator(words: string[], separator: string): string[] {
  function splitWord(word: string) {
    return word.split(separator).filter((s) => s !== '');
  }
  return words.flatMap(splitWord);
}
```
