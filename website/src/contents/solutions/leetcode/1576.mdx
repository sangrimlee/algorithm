## Solution

```typescript filename="solution.ts" copy
export function modifyString(s: string): string {
  const n = s.length;
  const arr = [...s];
  const replacables = ['a', 'b', 'c'];
  for (let i = 0; i < n; i++) {
    if (arr[i] !== '?') continue;
    arr[i] = replacables.find((r) => r !== arr[i - 1] && r !== arr[i + 1])!;
  }
  return arr.join('');
}
```
