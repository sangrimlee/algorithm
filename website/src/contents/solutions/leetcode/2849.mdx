## Solution

```typescript filename="solution.ts" copy
export function isReachableAtTime(
  sx: number,
  sy: number,
  fx: number,
  fy: number,
  t: number,
): boolean {
  if (sx === fx && sy === fy && t === 1) {
    return false;
  }

  return Math.max(Math.abs(sx - fx), Math.abs(sy - fy)) <= t;
}
```
