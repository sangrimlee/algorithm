## Solution

```typescript filename="solution.ts" copy
export function arithmeticTriplets(nums: number[], diff: number): number {
  const set = new Set(nums);
  return nums.filter((num) => set.has(num + diff) && set.has(num + diff * 2)).length;
}
```
