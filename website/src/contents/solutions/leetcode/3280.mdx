## Solution

```typescript showLineNumbers
export function convertDateToBinary(date: string): string {
  return date
    .split('-')
    .map((v) => parseInt(v).toString(2))
    .join('-');
}
```

### Complexity

- Time: `O(1)`
- Space: `O(1)`
