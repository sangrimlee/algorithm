## Solution

```typescript showLineNumbers
export function minStartValue(nums: number[]): number {
  let minValue = 0;
  let currentValue = 0;
  nums.forEach((num) => {
    currentValue += num;
    minValue = Math.min(minValue, currentValue);
  });
  return Math.max(-minValue, 0) + 1;
}
```
