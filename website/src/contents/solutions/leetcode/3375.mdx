## Solution

```typescript filename="solution.ts" copy
export function minOperations(nums: number[], k: number): number {
  if (nums.some((num) => num < k)) {
    return -1;
  }
  const set = new Set(nums);
  set.add(k);
  return set.size - 1;
}
```

### Complexity

- Time: $O(n)$
- Space: $O(n)$
