## Solution

```typescript filename="solution.ts" copy
export function minFlips(target: string): number {
  let answer = 0;
  let current = '0';
  for (const char of target) {
    if (current !== char) {
      current = char;
      answer += 1;
    }
  }
  return answer;
}
```

### Complexity

- Time: `O(N)`
- Space: `O(1)`
