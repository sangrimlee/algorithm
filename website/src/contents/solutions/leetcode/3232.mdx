## Solution

```typescript filename="solution.ts" copy
export function canAliceWin(nums: number[]): boolean {
  return nums.reduce((acc, num) => (num < 10 ? acc + num : acc - num), 0) !== 0;
}
```

### Complexity

- Time: `O(N)`
- Space: `O(1)`
