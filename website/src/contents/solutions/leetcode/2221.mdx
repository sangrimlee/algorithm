## Solution

```typescript filename="solution.ts" copy
export function triangularSum(nums: number[]): number {
  while (1 < nums.length) {
    for (let i = 0; i < nums.length; i++) {
      nums[i] = (nums[i] + nums[i + 1]) % 10;
    }
    nums.pop();
  }
  return nums[0];
}
```
