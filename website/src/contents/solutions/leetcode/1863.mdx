export function subsetXORSum(nums: number[]): number {
  function sumXOR(i: number, curr: number): number {
    if (i === nums.length) {
      return curr;
    }
    return sumXOR(i + 1, curr) + sumXOR(i + 1, curr ^ nums[i]);
  }
  return sumXOR(0, 0);
}
```

export function subsetXORSum(nums: number[]): number {
  const n = nums.length;
  const totalOR = nums.reduce((prev, num) => prev | num, 0);
  return totalOR << (n - 1);
}
```
