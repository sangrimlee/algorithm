## Solution

```typescript filename="solution.ts" copy
export function kItemsWithMaximumSum(
  numOnes: number,
  numZeros: number,
  numNegOnes: number,
  k: number,
): number {
  return Math.min(k, numOnes) - Math.max(0, k - numOnes - numZeros);
}
```
