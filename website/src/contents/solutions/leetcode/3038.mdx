## Solution

```typescript showLineNumbers
export function maxOperations(nums: number[]): number {
  const n = nums.length;
  const sum = nums[0] + nums[1];

  let answer = 1;
  for (let i = 2; i < n - 1; i += 2) {
    if (sum !== nums[i] + nums[i + 1]) {
      return answer;
    }
    answer += 1;
  }
  return answer;
}
```
