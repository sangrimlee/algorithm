## Solution

```typescript filename="solution.ts" copy
export function sumZero(n: number): number[] {
  const answer = n % 2 === 0 ? [] : [0];
  for (let num = 1; num <= Math.floor(n / 2); num++) {
    answer.push(num, -num);
  }
  return answer;
}
```
